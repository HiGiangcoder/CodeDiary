    # O(n^2)

    hist = {}

    for i in set(arr):

        count = 0

        for j in arr:
            if j == i:
                count += 1

    def calculate_histogram(arr):

    # O(n)

    hist = {}

    for i in range(len(arr)):

        if arr[i] in hist:
            hist[arr[i]] += 1
        else:
            hist[arr[i]] = 1

    return hist
